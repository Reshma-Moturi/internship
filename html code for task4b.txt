        <input type="text" name="pattern" id="pattern">
        <br>
        <button type="button" onclick="highlightPattern()">Find Pattern</button>

    <!-- Add a new section for email validation -->
    <div id="emailValidationForm">
        <h2>Email Validation</h2>
        <form id="validateEmailForm" onsubmit="return validateEmail()">
            <label for="email">Enter Email:</label>
            <input type="text" name="email" id="email">
            <br>
            <button type="submit">Validate Email</button>
        </form>
        <div id="emailValidationResult"></div>
    </div>

    <script>
        function highlightPattern() {
            var textDiv = document.getElementById("text");
            var patternInput = document.getElementById("pattern");

            var text = textDiv.textContent;
            var pattern = patternInput.value;

            var regex = new RegExp(pattern, "g");
            var highlighted = text.replace(regex, function (match) {
                return '<span class="highlighted">' + match + '</span>';
            });

            textDiv.innerHTML = highlighted;
        }

        function validateEmail() {
            var emailInput = document.getElementById("email");
            var validationResultDiv = document.getElementById("emailValidationResult");

            var email = emailInput.value;
            var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

            if (emailRegex.test(email)) {
                validationResultDiv.innerHTML = "Email is valid!";
            } else {
                validationResultDiv.innerHTML = "Email is not valid!";
            }

            return false; // Prevent form submission for demonstration purposes
        }
    </script>
